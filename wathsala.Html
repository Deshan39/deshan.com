// ============================== // Motor Driver Pin Configuration // ============================== #define enA 10 // Enable pin for Right Motor #define in1 9 // Right Motor input 1 #define in2 8 // Right Motor input 2 #define enB 5 // Enable pin for Left Motor #define in3 7 // Left Motor input 1 #define in4 6 // Left Motor input 2 // ============================== // Sensors & Modules // ============================== #define ir_R A0 // Right IR sensor #define ir_F A1 // Front IR sensor #define ir_L A2 // Left IR sensor #define servo A4 // Servo pin #define pump A5 // Pump pin int Speed = 160; // Motor speed (0-255) int s1, s2, s3; void setup() { Serial.begin(9600); // Sensor pins pinMode(ir_R, INPUT); pinMode(ir_F, INPUT); pinMode(ir_L, INPUT); // Motor pins pinMode(enA, OUTPUT); pinMode(in1, OUTPUT); pinMode(in2, OUTPUT); pinMode(enB, OUTPUT); pinMode(in3, OUTPUT); pinMode(in4, OUTPUT); // Extra modules pinMode(servo, OUTPUT); pinMode(pump, OUTPUT); // Initial servo sweep for (int angle = 90; angle <= 140; angle += 5) { servoPulse(servo, angle); } for (int angle = 140; angle >= 40; angle -= 5) { servoPulse(servo, angle); } for (int angle = 40; angle <= 95; angle += 5) { servoPulse(servo, angle); } // Set initial motor speed analogWrite(enA, Speed); analogWrite(enB, Speed); delay(500); } void loop() { // Read sensors s1 = analogRead(ir_R); s2 = analogRead(ir_F); s3 = analogRead(ir_L); // Print sensor values Serial.print(s1); Serial.print("\t"); Serial.print(s2); Serial.print("\t"); Serial.println(s3); delay(50); // ============================== // Fire / Obstacle Detection Logic // ============================== if (s1 < 250) { // Fire on right side Stop(); digitalWrite(pump, HIGH); for (int angle = 90; angle >= 40; angle -= 3) servoPulse(servo, angle); for (int angle = 40; angle <= 90; angle += 3) servoPulse(servo, angle); } else if (s2 < 350) { // Fire in front Stop(); digitalWrite(pump, HIGH); for (int angle = 90; angle <= 140; angle += 3) servoPulse(servo, angle); for (int angle = 140; angle >= 40; angle -= 3) servoPulse(servo, angle); for (int angle = 40; angle <= 90; angle += 3) servoPulse(servo, angle); } else if (s3 < 250) { // Fire on left side Stop(); digitalWrite(pump, HIGH); for (int angle = 90; angle <= 140; angle += 3) servoPulse(servo, angle); for (int angle = 140; angle >= 90; angle -= 3) servoPulse(servo, angle); } else if (s1 >= 251 && s1 <= 700) { // Obstacle right digitalWrite(pump, LOW); backword(); delay(100); turnRight(); delay(200); } else if (s2 >= 251 && s2 <= 800) { // Clear front digitalWrite(pump, LOW); forword(); } else if (s3 >= 251 && s3 <= 700) { // Obstacle left digitalWrite(pump, LOW); backword(); delay(100); turnLeft(); delay(200); } else { digitalWrite(pump, LOW); Stop(); } delay(10); } // ============================== // Servo Control Function // ============================== void servoPulse(int pin, int angle) { int pwm = (angle * 11) + 500; // Convert angle to microseconds digitalWrite(pin, HIGH); delayMicroseconds(pwm); digitalWrite(pin, LOW); delay(50); // Refresh cycle for servo } // ============================== // Motor Control Functions // ============================== void forword() { // Move forward digitalWrite(in1, HIGH); digitalWrite(in2, LOW); digitalWrite(in3, HIGH); digitalWrite(in4, LOW); } void backword() { // Move backward digitalWrite(in1, LOW); digitalWrite(in2, HIGH); digitalWrite(in3, LOW); digitalWrite(in4, HIGH); } void turnRight() { // Rotate right digitalWrite(in1, HIGH); digitalWrite(in2, LOW); digitalWrite(in3, LOW); digitalWrite(in4, HIGH); } void turnLeft() { // Rotate left digitalWrite(in1, LOW); digitalWrite(in2, HIGH); digitalWrite(in3, HIGH); digitalWrite(in4, LOW); } void Stop() { // Stop motors digitalWrite(in1, LOW); digitalWrite(in2, LOW); digitalWrite(in3, LOW); digitalWrite(in4, LOW); }
